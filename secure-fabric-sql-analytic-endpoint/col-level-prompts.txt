~~ AI Solution 2 ~~

CREATE VIEW [data_mart].[vw_FactSales_Masked]
AS
SELECT 
    [InvoiceId],
    [CustomerId],
    [InvoiceDate],
    [BillingAddress],
    [BillingCity],
    [BillingState],
    [BillingCountry],
    [BillingPostalCode],
    -- Masking logic: returns value if in a sales role, otherwise NULL
    CASE 
        WHEN IS_ROLEMEMBER('SalesRole01') = 1 or IS_ROLEMEMBER('SalesRole02') = 1
        THEN [Total] 
        ELSE NULL 
    END AS [Total],
    [InvoiceLineId],
    [TrackId],
    CASE 
        WHEN IS_ROLEMEMBER('SalesRole01') = 1 or IS_ROLEMEMBER('SalesRole02') = 1
        THEN [UnitPrice] 
        ELSE NULL 
    END AS [UnitPrice],
    [Quantity]
FROM [data_mart].[FactSales];
GO

-- Show rights
SELECT 
    USER_NAME(grantee_principal_id) AS [Principal_Name],
    permission_name AS [Permission],
    state_desc AS [Action_Type], -- GRANT, DENY, etc.
    COL_NAME(major_id, minor_id) AS [Column_Name], -- Null if table-level
    class_desc AS [Object_Class]
FROM sys.database_permissions
WHERE major_id = OBJECT_ID('[data_mart].[FactSales]')
ORDER BY [Principal_Name], [Action_Type];

-- Show memberships
SELECT 
    DP1.name AS [DatabaseRoleName], 
    DP2.name AS [DatabaseMemberName],
    DP2.type_desc AS [MemberType]
FROM sys.database_role_members AS DRM
INNER JOIN sys.database_principals AS DP1 
    ON DRM.role_principal_id = DP1.principal_id
INNER JOIN sys.database_principals AS DP2 
    ON DRM.member_principal_id = DP2.principal_id
ORDER BY [DatabaseRoleName], [DatabaseMemberName];
