-- Columns with mask
SELECT 
    t.name AS [Table_Name],
    mc.name AS [Column_Name],
    mc.masking_function AS [Mask_Function],
    mc.is_masked
FROM sys.masked_columns AS mc
JOIN sys.tables AS t ON mc.object_id = t.object_id
WHERE t.name = 'FactSales';
