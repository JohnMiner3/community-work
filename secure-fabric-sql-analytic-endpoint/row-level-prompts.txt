/* 1. Drop the Filter Function */
CREATE FUNCTION [data_mart].[fn_CountrySecurityPredicate](@Country AS NVARCHAR(MAX))
    RETURNS TABLE
WITH SCHEMABINDING
AS
RETURN SELECT 1 AS fn_security_predicate_result
WHERE 
    -- North America Access
    (IS_ROLEMEMBER('SalesRole01') = 1 AND @Country IN ('Canada', 'USA'))
    OR 
    -- Other Countries Access
    (IS_ROLEMEMBER('SalesRole02') = 1 AND @Country NOT IN ('Canada', 'USA'))
    OR
    -- Admin/Other User Access (Unrestricted)
    (IS_ROLEMEMBER('SalesRole01') = 0 AND IS_ROLEMEMBER('SalesRole02') = 0);
GO


/* 2. Create the Security Policy */
